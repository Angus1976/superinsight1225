## 1. JWT 认证测试程序

### 测试 1.1: Token 生成

```python
# 测试代码位置: tests/test_security_service.py

def test_jwt_token_generation():
    jwt_manager = JWTManager()
    
    # 生成 token
    token = jwt_manager.create_token(
        user_id="test_user_123",
        roles=["admin", "user"]
    )
    
    # 验证 token 格式
    assert isinstance(token, str)
    assert len(token.split('.')) == 3  # JWT 三部分
    
    # 解码验证内容
    payload = jwt_manager.validate_token(token)
    assert payload.user_id == "test_user_123"
    assert "admin" in payload.roles
```

### 测试 1.2: Token 验证

```python
def test_jwt_token_validation():
    jwt_manager = JWTManager()
    token = jwt_manager.create_token("user1", ["user"])
    
    # 有效 token 应该通过验证
    payload = jwt_manager.validate_token(token)
    assert payload.user_id == "user1"
    
    # 无效 token 应该失败
    with pytest.raises(ValueError):
        jwt_manager.validate_token("invalid.token.here")
```

### 测试 1.3: Token 过期

```python
def test_jwt_token_expiration():
    jwt_manager = JWTManager()
    
    # 创建 1 秒过期的 token
    token = jwt_manager.create_token(
        user_id="user1",
        roles=["user"],
        expiry_seconds=1
    )
    
    # 立即验证应该成功
    payload = jwt_manager.validate_token(token)
    assert payload.user_id == "user1"
    
    # 等待过期
    time.sleep(2)
    
    # 过期后验证应该失败
    with pytest.raises(ValueError, match="expired"):
        jwt_manager.validate_token(token)
```


## 2. 数据加密测试程序

### 测试 2.1: 加密和解密

```python
def test_data_encryption_decryption():
    encryptor = DataEncryptor()
    
    # 测试数据
    plaintext = "敏感数据：用户身份证号 123456789012345678"
    
    # 加密
    ciphertext = encryptor.encrypt(plaintext)
    assert ciphertext != plaintext
    assert len(ciphertext) > 0
    
    # 解密
    decrypted = encryptor.decrypt(ciphertext)
    assert decrypted == plaintext
```

### 测试 2.2: 加密密钥隔离

```python
def test_encryption_key_isolation():
    # 使用不同密钥的加密器
    encryptor1 = DataEncryptor()
    encryptor2 = DataEncryptor()
    
    plaintext = "secret data"
    
    # 用密钥1加密
    ciphertext = encryptor1.encrypt(plaintext)
    
    # 用密钥2解密应该失败
    with pytest.raises(ValueError, match="Decryption failed"):
        encryptor2.decrypt(ciphertext)
```

```python
def test_xss_sanitization():
    # 测试各种 XSS 攻击模式
    test_cases = [
        ("<script>alert('xss')</script>", "alert('xss')"),
        ("<img src=x onerror=alert(1)>", ""),
        ("<a href='javascript:alert(1)'>click</a>", "click"),
        ("<<SCRIPT>alert('XSS');//<</SCRIPT>", "alert('XSS');//"),
    ]
    
    for malicious_input, expected_output in test_cases:
        result = sanitize_input(malicious_input, strip_html=True)
        assert result.value == expected_output
        assert result.was_modified
```


### 测试 3.2: SQL 注入防护

```python
def test_sql_injection_sanitization():
    # 测试 SQL 注入模式
    test_cases = [
        "'; DROP TABLE users; --",
        "1' OR '1'='1",
        "admin'--",
        "' UNION SELECT * FROM passwords--",
    ]
    
    for sql_injection in test_cases:
        result = sanitize_input(sql_injection)
        # 验证危险字符被转义或移除
        assert "--" not in result.value or result.was_modified
        assert "DROP" not in result.value.upper() or result.was_modified
```

### 测试 3.3: 长度限制

```python
def test_input_length_limit():
```

```python
def test_dict_sanitization():
    malicious_dict = {
        "name": "<script>alert('xss')</script>",
        "email": "user@example.com",
        "nested": {
            "comment": "<img src=x onerror=alert(1)>"
        },
        "tags": ["<script>", "normal", "<b>bold</b>"]

    assert "<img" not in sanitized["nested"]["comment"]
```

```python
async def test_rate_limiting():
    # 配置: 每分钟最多 5 个请求
    rate_limiter = RateLimiter(max_requests=5, window_seconds=60)
    
    client_ip = "192.168.1.100"
    
    # 前 5 个请求应该成功
    for i in range(5):
        assert rate_limiter.check_limit(client_ip) == True
    
    # 第 6 个请求应该被拒绝
    assert rate_limiter.check_limit(client_ip) == False
```


### 测试 4.2: 速率限制重置

```python
async def test_rate_limit_reset():
    rate_limiter = RateLimiter(max_requests=3, window_seconds=2)
    
    client_ip = "192.168.1.100"
    
    # 用完配额
    for i in range(3):
        assert rate_limiter.check_limit(client_ip) == True
    
    # 应该被限制
    assert rate_limiter.check_limit(client_ip) == False
    
    # 等待窗口重置
    await asyncio.sleep(3)
    
    # 应该恢复
```

```python
def test_audit_log_filtering():

    # 创建多个日志
    for i in range(10):
        audit_logger.log_operation(
            operation_type=OperationType.READ if i % 2 == 0 else OperationType.UPDATE,
            user_id=f"user{i % 3}",
            resource_type=ResourceType.SAMPLE,
            resource_id=f"sample{i}",
            action=Action.VIEW,
            result=OperationResult.SUCCESS,
            duration=100 + i
        )
    
    # 按用户过滤
    user0_logs = audit_logger.get_audit_log(user_id="user0")
    assert all(log.user_id == "user0" for log in user0_logs)
    
    # 按操作类型过滤
    read_logs = audit_logger.get_audit_log(operation_type=OperationType.READ)
    assert all(log.operation_type == OperationType.READ for log in read_logs)
```

## 6. 集成安全测试

### 测试 6.1: 端到端认证流程

```python
async def test_e2e_authentication():
    # 1. 用户登录获取 token
    response = await client.post("/api/auth/login", json={
        "username": "testuser",
        "password": "securepass123"
    })
    assert response.status_code == 200
    token = response.json()["token"]
    
    # 2. 使用 token 访问受保护资源
    response = await client.get(
        "/api/samples",
        headers={"Authorization": f"Bearer {token}"}
    )
    assert response.status_code == 200
    
    # 3. 无 token 访问应该失败
    response = await client.get("/api/samples")
    assert response.status_code == 401
```

### 测试 6.2: 权限检查集成

```python
async def test_permission_enforcement():
    # 普通用户 token
    user_token = create_test_token(user_id="user1", roles=["user"])
    
    # 管理员 token
    admin_token = create_test_token(user_id="admin1", roles=["admin"])
    
    # 普通用户尝试删除（应该失败）
    response = await client.delete(
        "/api/samples/sample123",
        headers={"Authorization": f"Bearer {user_token}"}
    )
    assert response.status_code == 403
    
    # 管理员删除（应该成功）
    response = await client.delete(
        "/api/samples/sample123",
        headers={"Authorization": f"Bearer {admin_token}"}
    )
```
