#### 1.2 增强实现

**步骤 1**: 创建 Label Studio 项目同步服务

```typescript
// frontend/src/services/labelStudioService.ts

export const labelStudioService = {
  // 获取所有项目
  async getAllProjects() {
    const response = await apiClient.get('/api/label-studio/projects');
    return response.data.results || [];
  },
  
  // 获取项目详情
  async getProject(projectId: number) {
    const response = await apiClient.get(`/api/label-studio/projects/${projectId}`);
    return response.data;
  },
  
  // 获取项目任务列表
  async getProjectTasks(projectId: number) {
    const response = await apiClient.get(`/api/label-studio/projects/${projectId}/tasks`);
    return response.data.tasks || [];
  },
  
  // 同步单个任务
  async syncTask(taskId: string, projectId: number) {
    // 1. 获取 Label Studio 项目信息
    const project = await this.getProject(projectId);
    
    // 2. 获取项目任务列表
    const lsTasks = await this.getProjectTasks(projectId);
    
    // 3. 计算进度
    const totalTasks = lsTasks.length;
    const completedTasks = lsTasks.filter(t => t.is_labeled).length;
    const progress = totalTasks > 0 
      ? Math.round((completedTasks / totalTasks) * 100) 
      : 0;
    
    // 4. 更新本地任务
    const response = await apiClient.patch(`/api/tasks/${taskId}`, {
      total_items: totalTasks,
      completed_items: completedTasks,
      progress: progress,
      label_studio_sync_status: 'synced',
      label_studio_last_sync: new Date().toISOString(),
    });
    
    return response.data;
  },
  
  // 批量同步任务
  async syncAllTasks(tasks: Task[]) {
    const results = {
      success: [] as string[],
      failed: [] as { taskId: string; error: string }[],
    };
    
    for (const task of tasks) {
      if (!task.label_studio_project_id) {
        continue;
      }
      
      try {
        await this.syncTask(task.id, task.label_studio_project_id);
        results.success.push(task.id);
      } catch (error) {
        results.failed.push({
          taskId: task.id,
          error: error instanceof Error ? error.message : 'Unknown error',
        });
      }
    }
    
    return results;
  },
};
```

**步骤 2**: 更新任务列表页面的刷新逻辑

```typescript
// frontend/src/pages/Tasks/index.tsx

const handleSyncAllTasks = async () => {
  const tasksToSync = (data?.items || []).filter(
    task => task.label_studio_project_id
  );
  
  if (tasksToSync.length === 0) {
    message.info('没有需要同步的任务');
    return;
  }
  
  const hide = message.loading(
    `正在同步 ${tasksToSync.length} 个任务...`, 
    0
  );
  
  try {
    // 使用新的同步服务
    const results = await labelStudioService.syncAllTasks(tasksToSync);
    
    hide();
    
    if (results.failed.length === 0) {
      message.success(`成功同步 ${results.success.length} 个任务`);
    } else {
      message.warning(
        `同步完成：成功 ${results.success.length} 个，失败 ${results.failed.length} 个`
      );
      
      // 显示失败详情
      console.error('Sync failures:', results.failed);
    }
    
    // 刷新列表
    refetch();
    actionRef.current?.reload();
    
  } catch (error) {
    hide();
    console.error('Sync failed:', error);
    message.error('同步失败，请重试');
  }
};
```

// 使用 Modal 显示同步进度
const handleSyncAllTasksWithProgress = async () => {
  const tasksToSync = (data?.items || []).filter(
    task => task.label_studio_project_id
  );
  
  if (tasksToSync.length === 0) {
    message.info('没有需要同步的任务');

  }
  
  // 创建进度 Modal
  const progressModal = Modal.info({
    title: '同步进度',
    content: (
      <div>
        <Progress percent={0} />
        <p>正在同步任务 0 / {tasksToSync.length}</p>
      </div>
    ),
    okButtonProps: { style: { display: 'none' } },
  });
  
  try {
    let completed = 0;
    const results = {
      success: [] as string[],
      failed: [] as { taskId: string; error: string }[],
    };
    
    for (const task of tasksToSync) {
      try {
        await labelStudioService.syncTask(
          task.id, 
          task.label_studio_project_id!
        );
        results.success.push(task.id);
      } catch (error) {
        results.failed.push({
          taskId: task.id,
          error: error instanceof Error ? error.message : 'Unknown error',
        });
      }
      
      completed++;
      const percent = Math.round((completed / tasksToSync.length) * 100);
      
      // 更新进度
      progressModal.update({
        content: (
          <div>
            <Progress percent={percent} />
            <p>正在同步任务 {completed} / {tasksToSync.length}</p>
          </div>
        ),
      });
    }
    
    // 关闭进度 Modal
    progressModal.destroy();
    
    // 显示结果
    if (results.failed.length === 0) {
      message.success(`成功同步 ${results.success.length} 个任务`);
    } else {
      Modal.warning({
        title: '同步完成',
        content: (
          <div>
            <p>成功: {results.success.length} 个</p>
            <p>失败: {results.failed.length} 个</p>
            {results.failed.length > 0 && (
              <div style={{ marginTop: 16 }}>
                <p>失败的任务:</p>
                <ul>
                  {results.failed.map(f => (
                    <li key={f.taskId}>
                      任务 {f.taskId}: {f.error}
                    </li>
                  ))}
                </ul>
              </div>
            )}
          </div>
        ),
      });
    }
    
    // 刷新列表
    refetch();
    actionRef.current?.reload();
    
  } catch (error) {
    progressModal.destroy();
    console.error('Sync failed:', error);
    message.error('同步失败，请重试');
  }
};

```typescript
// frontend/src/services/exportService.ts

import * as XLSX from 'xlsx';
import type { Task } from '@/types';

export const exportService = {
  // 导出为 CSV
  exportToCSV(tasks: Task[], filename?: string) {
    const headers = [
      'ID', '名称', '状态', '优先级', '标注类型',
      '进度', '完成数', '总数', '分配人', '创建时间', '截止日期'
    ];
    
    const rows = tasks.map(task => [
      task.id,
      task.name,
      task.status,
      task.priority,
      task.annotation_type,
      `${task.progress}%`,
      task.completed_items,
      task.total_items,
      task.assignee_name || '',
      new Date(task.created_at).toLocaleDateString(),
      task.due_date ? new Date(task.due_date).toLocaleDateString() : ''
    ]);
    
    const csvContent = [
      headers.join(','),
      ...rows.map(row => row.map(cell => `"${cell}"`).join(','))
    ].join('\n');
    
    this.downloadFile(
      csvContent, 
      filename || `tasks_${new Date().toISOString().split('T')[0]}.csv`,
      'text/csv;charset=utf-8;'
    );
  },
  
  // 导出为 JSON
  async exportToJSON(tasks: Task[], includeAnnotations: boolean = false, filename?: string) {
    let exportData = tasks;
    
    if (includeAnnotations) {
      // 获取标注结果
      exportData = await Promise.all(
        tasks.map(async task => {
          let annotations = [];
          if (task.label_studio_project_id) {
            try {
              const response = await apiClient.get(
                `/api/label-studio/projects/${task.label_studio_project_id}/export`
              );
              annotations = response.data;
            } catch (error) {
              console.error(`Failed to export annotations for task ${task.id}`);
            }
          }
          
          return {
            ...task,
            annotations
          };
        })
      );
    }
    
    const jsonContent = JSON.stringify(exportData, null, 2);
    
    this.downloadFile(
      jsonContent,
      filename || `tasks_${new Date().toISOString().split('T')[0]}.json`,
      'application/json'
    );
  },
  
  // 导出为 Excel
  exportToExcel(tasks: Task[], filename?: string) {
    // Sheet 1: 任务列表
    const taskSheet = XLSX.utils.json_to_sheet(
      tasks.map(task => ({
        'ID': task.id,
        '名称': task.name,
        '状态': task.status,
        '优先级': task.priority,
        '标注类型': task.annotation_type,
        '进度': `${task.progress}%`,
        '完成数': task.completed_items,
        '总数': task.total_items,
        '分配人': task.assignee_name || '',
        '创建时间': new Date(task.created_at).toLocaleDateString(),
        '截止日期': task.due_date ? new Date(task.due_date).toLocaleDateString() : ''
      }))
    );
    
    // Sheet 2: 统计信息
    const statsSheet = XLSX.utils.json_to_sheet([
      { '指标': '总任务数', '数值': tasks.length },
      { '指标': '待处理', '数值': tasks.filter(t => t.status === 'pending').length },
      { '指标': '进行中', '数值': tasks.filter(t => t.status === 'in_progress').length },
      { '指标': '已完成', '数值': tasks.filter(t => t.status === 'completed').length },
      { '指标': '已取消', '数值': tasks.filter(t => t.status === 'cancelled').length },
      { '指标': '平均进度', '数值': `${Math.round(tasks.reduce((sum, t) => sum + t.progress, 0) / tasks.length)}%` }
    ]);
    
    // 创建工作簿
    const workbook = XLSX.utils.book_new();
    XLSX.utils.book_append_sheet(workbook, taskSheet, '任务列表');
    XLSX.utils.book_append_sheet(workbook, statsSheet, '统计信息');
    
    // 导出文件
    XLSX.writeFile(
      workbook, 
      filename || `tasks_${new Date().toISOString().split('T')[0]}.xlsx`
    );
  },
  
  // 下载文件
  downloadFile(content: string, filename: string, mimeType: string) {
    const blob = new Blob([content], { type: mimeType });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  }
};
```

const handleExportTasks = () => {
  const tasksToExport = selectedRowKeys.length > 0

          <Space direction="vertical">
            <Radio value="csv">CSV (简单数据)</Radio>
            <Radio value="json">JSON (完整数据)</Radio>
            <Radio value="excel">Excel (报表)</Radio>
          </Space>
        </Radio.Group>
        
        {exportFormat === 'json' && (
          <Checkbox
            checked={includeAnnotations}
            onChange={e => setIncludeAnnotations(e.target.checked)}
            style={{ marginTop: 16 }}
          >
            包含标注结果
          </Checkbox>
        )}
      </div>
    ),
    onOk: async () => {
      const hide = message.loading('正在导出...', 0);
      
      try {
        switch (exportFormat) {
          case 'csv':
            exportService.exportToCSV(tasksToExport);
            break;
          case 'json':
            await exportService.exportToJSON(tasksToExport, includeAnnotations);
            break;
          case 'excel':
            exportService.exportToExcel(tasksToExport);
            break;
        }
        
        hide();
        message.success('导出成功');
      } catch (error) {
        hide();
        message.error('导出失败');
      }
    }
  });
